lmmodel = function(data,brand,y)
{
  if(brand == "三星"){i = 2}
  if(brand == "苹果"){i = 4}
  if(brand == "HTC") {i = 6}
  if(brand == "华为"){i = 7}
  
  if(y == "满意度") {j = 6}
  if(y == "忠诚度") {j = 7}
  
  # i = 2 #品牌
  # j = 6 #6满意度，7忠诚度
  lm = lm(data[,j]~data[,3]+data[,4]+data[,5],data,which(data[,2]==i))
  summary(lm)
  plot(lm$residuals)
  text(1:sum(data[,2]==i),lm$residuals,1:sum(data[,2]==i),cex=0.8,pos=3,col="blue")#标记样本点编号
  return(summary(lm))
}
